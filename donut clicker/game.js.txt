let donuts = 0;
let autoDonutMachines = 0;
const autoDonutCost = 100;
const donutDisplay = document.getElementById('donutCount');
const autoCountDisplay = document.getElementById('autoCount');
const clickButton = document.getElementById('clickButton');
const buyAutoButton = document.getElementById('buyAuto');

// Update the donut display
function updateDonutDisplay() {
  donutDisplay.textContent = donuts;
}

// Update the auto machine display and button
function updateAutoMachineDisplay() {
  autoCountDisplay.textContent = autoDonutMachines;
  buyAutoButton.disabled = donuts < autoDonutCost; // Disable the button if you don't have enough donuts
}

// Handle the donut click button
clickButton.addEventListener('click', function() {
  donuts++;
  updateDonutDisplay();
  updateAutoMachineDisplay();
});

// Handle purchasing the auto donut machine
buyAutoButton.addEventListener('click', function() {
  if (donuts >= autoDonutCost) {
    donuts -= autoDonutCost;
    autoDonutMachines++;
    updateDonutDisplay();
    updateAutoMachineDisplay();
  }
});

// Automatically generate donuts from machines every second
setInterval(function() {
  donuts += autoDonutMachines;
  updateDonutDisplay();
}, 1000); // 1000ms = 1 second

// Initialize the game display
updateDonutDisplay();
updateAutoMachineDisplay();
